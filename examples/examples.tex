	\item[] NSMomentumTimeDerivative
	\item[] NSMomentum
	\item[] MatINSTemperatureTimeDerivative

	\item[] NSTemperatureVD
\end{description}

Equations involved:
\begin{equation}
\rho (\vec{v}.\nabla)\vec{v} + \nabla p - \rho g \check{j} - \mu \nabla^{2}\vec{v} = 0
\end{equation}

\nabla.\vec{v}=0

	\item[] $T('left')=100$, $T('right')=100$, and $T('top')=0$.
